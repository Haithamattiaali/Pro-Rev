const ETLService = require('./etl.service');
const { createTestDatabase, seedTestData } = require('../test/utils');
const XLSX = require('xlsx');

// Mock the database modules
jest.mock('../database/db-wrapper');
jest.mock('../database/persistent-db');

describe('ETLService', () => {
  let etlService;
  let testDb;
  const mockDbWrapper = require('../database/db-wrapper');
  const mockPersistentDb = require('../database/persistent-db');

  beforeEach(() => {
    testDb = createTestDatabase();
    
    // Configure mocks to use test database
    mockDbWrapper.run = jest.fn((sql, params) => testDb.prepare(sql).run(params));
    mockDbWrapper.get = jest.fn((sql, params) => testDb.prepare(sql).get(params));
    mockDbWrapper.all = jest.fn((sql, params) => testDb.prepare(sql).all(params));
    mockDbWrapper.transaction = jest.fn((fn) => testDb.transaction(fn));
    
    mockPersistentDb.run = jest.fn((sql, params) => testDb.prepare(sql).run(params));
    mockPersistentDb.prepare = jest.fn((sql) => testDb.prepare(sql));
    mockPersistentDb.transaction = jest.fn((fn) => testDb.transaction(fn));
    
    etlService = new ETLService();
  });

  afterEach(() => {
    if (testDb) {
      testDb.close();
    }
  });

  describe('parseExcelFile', () => {
    it('should parse valid Excel file buffer', () => {
      const data = [
        ['Customer', 'ServiceType', 'Year', 'Month', 'Cost', 'Target', 'Revenue', 'Receivables Collected'],
        ['Customer A', 'Transportation', 2025, 'Jan', 100000, 150000, 140000, 130000],
        ['Customer B', 'Warehouses', 2025, 'Feb', 50000, 80000, 75000, 70000]
      ];

      const ws = XLSX.utils.aoa_to_sheet(data);
      const wb = XLSX.utils.book_new();
      XLSX.utils.book_append_sheet(wb, ws, 'Sheet1');
      const buffer = XLSX.write(wb, { bookType: 'xlsx', type: 'buffer' });

      const result = etlService.parseExcelFile(buffer);

      expect(result).toHaveLength(2);
      expect(result[0]).toMatchObject({
        Customer: 'Customer A',
        ServiceType: 'Transportation',
        Year: 2025,
        Month: 'Jan'
      });
    });

    it('should throw error for empty file', () => {
      const ws = XLSX.utils.aoa_to_sheet([]);
      const wb = XLSX.utils.book_new();
      XLSX.utils.book_append_sheet(wb, ws, 'Sheet1');
      const buffer = XLSX.write(wb, { bookType: 'xlsx', type: 'buffer' });

      expect(() => etlService.parseExcelFile(buffer)).toThrow('Excel file is empty');
    });

    it('should throw error for invalid file format', () => {
      const invalidBuffer = Buffer.from('invalid excel data');
      expect(() => etlService.parseExcelFile(invalidBuffer)).toThrow();
    });
  });

  describe('validateData', () => {
    it('should validate correct data structure', () => {
      const validData = [
        {
          Customer: 'Customer A',
          ServiceType: 'Transportation',
          Year: 2025,
          Month: 'Jan',
          Cost: 100000,
          Target: 150000,
          Revenue: 140000,
          'Receivables Collected': 130000
        }
      ];

      const result = etlService.validateData(validData);
      expect(result.isValid).toBe(true);
      expect(result.errors).toHaveLength(0);
    });

    it('should detect missing required columns', () => {
      const invalidData = [
        {
          Customer: 'Customer A',
          Year: 2025,
          Month: 'Jan'
        }
      ];

      const result = etlService.validateData(invalidData);
      expect(result.isValid).toBe(false);
      expect(result.errors).toContain('Missing required columns: ServiceType, Cost, Target, Revenue, Receivables Collected');
    });

    it('should validate data types', () => {
      const invalidData = [
        {
          Customer: 'Customer A',
          ServiceType: 'Transportation',
          Year: 'invalid',
          Month: 'Jan',
          Cost: 'not a number',
          Target: 150000,
          Revenue: 140000,
          'Receivables Collected': 130000
        }
      ];

      const result = etlService.validateData(invalidData);
      expect(result.isValid).toBe(false);
      expect(result.errors.length).toBeGreaterThan(0);
    });

    it('should validate month values', () => {
      const invalidData = [
        {
          Customer: 'Customer A',
          ServiceType: 'Transportation',
          Year: 2025,
          Month: 'InvalidMonth',
          Cost: 100000,
          Target: 150000,
          Revenue: 140000,
          'Receivables Collected': 130000
        }
      ];

      const result = etlService.validateData(invalidData);
      expect(result.isValid).toBe(false);
      expect(result.errors[0]).toContain('Invalid month');
    });

    it('should validate service types', () => {
      const invalidData = [
        {
          Customer: 'Customer A',
          ServiceType: 'InvalidService',
          Year: 2025,
          Month: 'Jan',
          Cost: 100000,
          Target: 150000,
          Revenue: 140000,
          'Receivables Collected': 130000
        }
      ];

      const result = etlService.validateData(invalidData);
      expect(result.isValid).toBe(false);
      expect(result.errors[0]).toContain('Invalid service type');
    });
  });

  describe('transformData', () => {
    it('should transform valid data correctly', () => {
      const inputData = [
        {
          Customer: 'Customer A',
          ServiceType: 'Transportation',
          Year: 2025,
          Month: 'Jan',
          Cost: 100000,
          Target: 150000,
          Revenue: 140000,
          'Receivables Collected': 130000
        }
      ];

      const result = etlService.transformData(inputData);

      expect(result).toHaveLength(1);
      expect(result[0]).toMatchObject({
        customer: 'Customer A',
        service_type: 'Transportation',
        year: 2025,
        month: 'Jan',
        cost: 100000,
        target: 150000,
        revenue: 140000,
        receivables_collected: 130000
      });
    });

    it('should handle pro-rating for current month', () => {
      const currentMonth = new Date().toLocaleString('en-US', { month: 'short' });
      const currentYear = new Date().getFullYear();
      
      const inputData = [
        {
          Customer: 'Customer A',
          ServiceType: 'Transportation',
          Year: currentYear,
          Month: currentMonth,
          Cost: 100000,
          Target: 150000,
          Revenue: 140000,
          'Receivables Collected': 130000
        }
      ];

      const result = etlService.transformData(inputData);
      
      // Check that pro-rating was applied
      expect(result[0].original_target).toBe(150000);
      expect(result[0].original_cost).toBe(100000);
      expect(result[0].target).toBeLessThanOrEqual(150000);
      expect(result[0].cost).toBeLessThanOrEqual(100000);
    });

    it('should normalize customer names', () => {
      const inputData = [
        {
          Customer: '  customer a  ',
          ServiceType: 'Transportation',
          Year: 2025,
          Month: 'Jan',
          Cost: 100000,
          Target: 150000,
          Revenue: 140000,
          'Receivables Collected': 130000
        }
      ];

      const result = etlService.transformData(inputData);
      expect(result[0].customer).toBe('Customer A');
    });
  });

  describe('loadData', () => {
    it('should insert new data successfully', () => {
      const testData = [
        {
          customer: 'New Customer',
          service_type: 'Transportation',
          year: 2025,
          month: 'Mar',
          cost: 100000,
          target: 150000,
          revenue: 140000,
          receivables_collected: 130000
        }
      ];

      const result = etlService.loadData(testData);
      expect(result.inserted).toBe(1);
      expect(result.updated).toBe(0);
      expect(result.errors).toHaveLength(0);

      // Verify data was inserted
      const stmt = testDb.prepare('SELECT * FROM revenue_data WHERE customer = ?');
      const inserted = stmt.all('New Customer');
      expect(inserted).toHaveLength(1);
    });

    it('should update existing data', () => {
      // First insert
      seedTestData(testDb);

      const updateData = [
        {
          customer: 'Test Customer A',
          service_type: 'Transportation',
          year: 2025,
          month: 'Jan',
          cost: 110000,
          target: 160000,
          revenue: 150000,
          receivables_collected: 140000
        }
      ];

      const result = etlService.loadData(updateData);
      expect(result.inserted).toBe(0);
      expect(result.updated).toBe(1);
      expect(result.errors).toHaveLength(0);

      // Verify data was updated
      const stmt = testDb.prepare('SELECT * FROM revenue_data WHERE customer = ? AND month = ?');
      const updated = stmt.get('Test Customer A', 'Jan');
      expect(updated.revenue).toBe(150000);
    });

    it('should handle database errors gracefully', () => {
      // Close database to simulate error
      testDb.close();

      const testData = [
        {
          customer: 'Test Customer',
          service_type: 'Transportation',
          year: 2025,
          month: 'Jan',
          cost: 100000,
          target: 150000,
          revenue: 140000,
          receivables_collected: 130000
        }
      ];

      expect(() => etlService.loadData(testData)).toThrow();
    });
  });

  describe('process (integration)', () => {
    it('should process valid Excel file end-to-end', () => {
      const data = [
        ['Customer', 'ServiceType', 'Year', 'Month', 'Cost', 'Target', 'Revenue', 'Receivables Collected'],
        ['Process Test', 'Transportation', 2025, 'Apr', 100000, 150000, 140000, 130000]
      ];

      const ws = XLSX.utils.aoa_to_sheet(data);
      const wb = XLSX.utils.book_new();
      XLSX.utils.book_append_sheet(wb, ws, 'Sheet1');
      const buffer = XLSX.write(wb, { bookType: 'xlsx', type: 'buffer' });

      const result = etlService.process(buffer);

      expect(result.success).toBe(true);
      expect(result.summary.totalRecords).toBe(1);
      expect(result.summary.inserted).toBe(1);
      expect(result.summary.updated).toBe(0);
      expect(result.summary.errors).toBe(0);
    });

    it('should return validation errors for invalid data', () => {
      const data = [
        ['Customer', 'ServiceType', 'Year', 'Month', 'Cost', 'Target', 'Revenue', 'Receivables Collected'],
        ['Invalid Test', 'InvalidService', 2025, 'Jan', 100000, 150000, 140000, 130000]
      ];

      const ws = XLSX.utils.aoa_to_sheet(data);
      const wb = XLSX.utils.book_new();
      XLSX.utils.book_append_sheet(wb, ws, 'Sheet1');
      const buffer = XLSX.write(wb, { bookType: 'xlsx', type: 'buffer' });

      const result = etlService.process(buffer);

      expect(result.success).toBe(false);
      expect(result.errors.length).toBeGreaterThan(0);
    });
  });
});
